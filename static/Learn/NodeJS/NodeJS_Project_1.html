Employee Attendance Page

Below is the code for a Node.js application that passes data of multiple employees to an HTML page, which resembles an attendance sheet. 


// app.js - Main file
const express = require('express');
const app = express();
const path = require('path');

// Sample employee data
const employees = [
    { id: 1, name: 'John', age: 30 },
    { id: 2, name: 'Alice', age: 25 },
    { id: 3, name: 'Bob', age: 35 },
    { id: 4, name: 'Eva', age: 28 },
    { id: 5, name: 'Michael', age: 32 },
    { id: 6, name: 'Emma', age: 27 },
    { id: 7, name: 'Ryan', age: 29 },
    { id: 8, name: 'Sophia', age: 31 },
    { id: 9, name: 'David', age: 26 },
    { id: 10, name: 'Olivia', age: 33 }
];

app.get('/', (req, res) => {
    res.sendFile(path.join(__dirname, 'index.html'));
});

app.get('/employees', (req, res) => {
    res.send(employees);
});

app.listen(3000, () => {
    console.log('Server is running on port 3000');
});
        


<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Employee Attendance</title>
    <link rel="stylesheet" href="styles.css">
</head>
<body>
    <h1>Employee Attendance Sheet</h1>
    <table>
        <tr>
            <th>ID</th>
            <th>Name</th>
            <th>Age</th>
        </tr>
        
    </table>
    <script>
        fetch('/employees')
            .then(response => response.json())
            .then(data => {
                const table = document.querySelector('table');
                data.forEach(employee => {
                    const row = table.insertRow();
                    row.innerHTML = `<td>${employee.id}</td><td>${employee.name}</td><td>${employee.age}</td>`;
                });
            });
    </script>
</body>
</html>

Explanation :

  • app.js: This file defines a Node.js application using Express. It contains an array of sample employee data and sets up routes to serve the HTML page and employee data.
  • index.html: This HTML page displays an employee attendance sheet. It dynamically fetches employee data from the server and populates the table with the retrieved data using JavaScript.

Passing Tables from 45 to 50

Below is the code for a Node.js application that passes tables with numbers from 45 to 50 to an HTML page. 


// app.js - Main file
const express = require('express');
const app = express();
const path = require('path');

app.get('/', (req, res) => {
   res.sendFile(path.join(__dirname, 'index.html'));
});

app.get('/tables', (req, res) => {
   const tables = [];
   for (let i = 45; i <= 50; i++) {
       const table = [];
       for (let j = 1; j <= 10; j++) {
           table.push(`${i} x ${j} = ${i * j}`);
       }
       tables.push(table);
   }
   res.json(tables);
});

app.listen(3000, () => {
   console.log('Server is running on port 3000');
});
       

<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>Tables from 45 to 50</title>
  <script src="https://cdn.jsdelivr.net/npm/axios/dist/axios.min.js"></script>
</head>
<body>
  <div id="tables"></div>
  <script>
      axios.get('/tables')
          .then(response => {
              const tables = response.data;
              const tablesContainer = document.getElementById('tables');
              tables.forEach(table => {
                  const tableElement = document.createElement('table');
                  table.forEach(row => {
                      const tr = document.createElement('tr');
                      const td = document.createElement('td');
                      td.textContent = row;
                      tr.appendChild(td);
                      tableElement.appendChild(tr);
                  });
                  tablesContainer.appendChild(tableElement);
              });
          })
          .catch(error => {
              console.error('Error fetching tables:', error);
          });
  </script>
</body>
</html>

Explanation :
This code defines a Node.js application using Express. It sets up routes to serve an HTML page and generate tables from 45 to 50. Each table contains multiplication results from 1 to 10.
